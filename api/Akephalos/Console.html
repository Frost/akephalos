def self.start
  require 'irb'

  begin
    require 'irb/completion'
  rescue Exception
    # No readline available, proceed anyway.
  end

  if ::File.exists? ".irbrc"
    ENV['IRBRC'] = ".irbrc"
  end

  IRB.start
end
