def set(value)
  if tag_name == 'textarea'
    native.value = value.to_s
  elsif tag_name == 'input' and type == 'radio'
    click
  elsif tag_name == 'input' and type == 'checkbox'
    if value != self['checked']
      click
    end
  elsif tag_name == 'input'
    native.value = value.to_s
  end
end
